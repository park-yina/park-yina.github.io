#include<iostream>
#include<queue>
using namespace std;
int map[3][3];
bool visit[3][3];
queue<pair<int, int>>q;
int dx[] = { 0,1 };
int dy[] = { 1,0 };//ì•„ë˜ì™€ ì™¼ìª½ìœ¼ë¡œ ì›€ì§ì´ëŠ” ê²ƒì„ ë‚˜íƒ€ë‚´ê¸° ìœ„í•˜ì—¬
int n;
int bfs(int x, int y) {
	q.push(make_pair(x, y));
	visit[0][0] = true;
	while (!q.empty()) {
		int x = q.front().first;
		int y = q.front().second;
		visit[x][y] = true;
		q.pop();
		int step = map[x][y];
		for (int i = 0; i < 2; i++) {
			int nx = x + step * dx[i];
			int ny = y + step * dy[i];
			if (!visit[nx][ny] && nx<n&&ny<n) {
				visit[nx][ny] = true;
				if (map[nx][ny] == -1) {
					return 1;
				}
				q.push(make_pair(nx, ny));
			}
			}
		}
	return 0;
	}
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);
	cin >> n;
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < n; j++) {
			cin >> map[i][j];
		}
	}
	int ans=bfs(0, 0);
	if (ans) {
		cout << "HaruHaru" << "\n";
	}
	else {
		cout << "Hing" << "\n";
	}
}
